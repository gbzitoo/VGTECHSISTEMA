CREATE DATABASE vgti_system;


CREATE TABLE LOGIN_VGTI(
VGTI_ID int(11) NOT NULL AUTO_INCREMENT,
VGTI_USUARIO VARCHAR(50) NOT NULL,
VGTI_SENHA VARCHAR(40) NOT NULL,
VGTI_NOME VARCHAR(40) NOT NULL,
VGTI_SETOR VARCHAR(40) NOT NULL,
PRIMARY KEY(VGTI_ID)
);

CREATE TABLE SERVICOS_VGTI(
VGTI_IDSERVICOS int(11) NOT NULL AUTO_INCREMENT,
VGTI_NOMEDOSERVICOS VARCHAR(50) NOT NULL,
VGTI_PRECO VARCHAR(40) NOT NULL,
PRIMARY KEY(VGTI_IDSERVICOS)
);

CREATE TABLE ATENDIMENTOS_VGTI(
VGTI_IDATENDIMENTO int(11) NOT NULL AUTO_INCREMENT,
VGTI_NOME VARCHAR(50) NOT NULL,
VGTI_EQUIPAMENTO VARCHAR(40) NOT NULL,
VGTI_SERVICO VARCHAR(40) NOT NULL,
VGTI_PRECO VARCHAR(40) NOT NULL,
PRIMARY KEY(VGTI_IDATENDIMENTO)
);

CREATE TABLE ATENDIMENTOSFECHADOS_VGTI(
VGTI_IDATENDIMENTO int(11) NOT NULL AUTO_INCREMENT,
VGTI_NOME VARCHAR(50) NOT NULL,
VGTI_EQUIPAMENTO VARCHAR(40) NOT NULL,
VGTI_SERVICO VARCHAR(40) NOT NULL,
VGTI_PRECO VARCHAR(40) NOT NULL,
PRIMARY KEY(VGTI_IDATENDIMENTO)
);
